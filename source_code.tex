\subsection{Definition}
\begin{verbatim}
sage: import slabbe.mult_cont_frac as mcf
sage: algo = mcf.Brun()
\end{verbatim}
\subsection{Invariant measure}
\begin{verbatim}

                                  ndivs=40,
                                  norm='1')
\end{verbatim}
